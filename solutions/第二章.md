std::vector<int> & re_insertion_sort(std::vector<int> & s){
    int i = 0, key;
    for(int j = 1; j < s.size(); ++j){
        key = s[j];
        i = j - 1;
        while(i >= 0 && s[i] < key){
            s[i + 1] = s[i];
            i = i - 1;
        }
        s[i + 1] = key;
    }
    return s;
}

std::vector<int> addition(std::vector<int> &A, std::vector<int> &B, int n){
    std::vector<int> C(n + 1, 0);
    int c = 0;
    for(int i = C.size() - 1; i >= 1; --i){
        C[i] = (c + A[i-1] + B[i-1]) % 2;
        c = (c + A[i-1] + B[i-1]) / 2;
    }
    C[0] = c;
    return C;
}
